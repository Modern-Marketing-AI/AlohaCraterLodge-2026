const RSS_URL = `https://volcanoes.usgs.gov/hans-public/api/feed/hvo`;

async function getVolcanoStatus() {
  try {
    const response = await fetch(RSS_URL);
    if (!response.ok) throw new Error('Network response was not ok');
    const text = await response.text();
    
    // Regex extracts the first headline from the USGS RSS feed
    const titleMatch = text.match(/<item>[\s\S]*?<title>(.*?)<\/title>/);
    return titleMatch ? titleMatch[1] : "The volcano is currently being monitored by the USGS.";
  } catch (error) {
    console.error("USGS Feed Error:", error);
    return "Status currently unavailable. Please check the live feed on our homepage.";
  }
}

// 1. Fetch the real-time status
const currentStatus = await getVolcanoStatus();
